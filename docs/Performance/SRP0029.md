## Description
  
Specify a window frame explicitly for window functions with ORDER BY to avoid implicit RANGE semantics.
  
## Summary
  
Specify a window frame in window functions with ORDER BY to avoid implicit RANGE framing.
  
### Examples
  
```sql
/* Default is Range UNBOUNDED PRECEDING ( Defining the frame )*/
Create Procedure dbo.ImplicitRangeWindow
as
Set nocount on
select sum(TestTableSSDT.Col1) over(partition by TestTableSSDT.IdCol
	                      order by TestTableSSDT.Col2
	                 ) as RollingBalance,
       row_number() over(partition by TestTableSSDT.IdCol
	                      order by TestTableSSDT.Col2 ) as Rown
  from dbo.TestTableSSDT
order by Col1,Col2;

-- SRP0029
```
```sql
/* Default is RANGE UNBOUNDED PRECEDING (defining the frame) */
CREATE PROCEDURE dbo.ImplicitRangeWindowSrp0029
AS
SET NOCOUNT ON;

SELECT SUM(tt.Col1) OVER (PARTITION BY tt.IdCol ORDER BY tt.Col2) AS RollingBalance,
       ROW_NUMBER() OVER (PARTITION BY tt.IdCol ORDER BY tt.Col2) AS Rown
FROM dbo.TestTableSSDT AS tt
ORDER BY tt.Col1, tt.Col2;

-- SRP0029
```
